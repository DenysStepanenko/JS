Основы JavaScript

Основы JavaScript включают переменные, типы данных, операторы, строки и условные конструкции. Эти концепции являются фундаментом для дальнейшего изучения языка.

Переменные и типы данных

В JavaScript существует три способа объявления переменных: let, const и устаревший var. 

// Этот код демонстрирует объявление переменных и основные типы данных в JavaScript.

// Выводим заголовок в консоль для лучшей читаемости
console.log("--- Типы данных ---");

// Объявление Переменных
// let: Используется для переменных, значение которых может меняться.
let message = "Привет";
// const: Используется для констант, значение которых не может быть изменено после присвоения.
const PI = 3.14;
// var: Устаревший способ объявления переменных. В современном коде рекомендуется использовать let и const.
var oldWay = "Устаревший способ";

// Основные типы данных в JavaScript:

// String (Строка): Последовательность символов.
let string = "Это строка";
// Number (Число): Целые числа и числа с плавающей точкой.
let number = 42;
// BigInt: Для очень больших целых чисел, которые не помещаются в стандартный Number.
let bigInt = 9007199254740991n;
// Boolean (Логический тип): Может принимать только два значения - true (истина) или false (ложь).
let boolean = true;
// Null: Специальное значение, означающее "ничего", "пусто" или "значение неизвестно".
let nullValue = null;
// Undefined: Значение, которое переменная получает по умолчанию, если ей ничего не присвоено.
let undefinedValue;
// Object (Объект): Коллекция свойств в формате ключ: значение.
let object = { name: "John", age: 30 };
// Array (Массив): Особый тип объекта, предназначенный для хранения упорядоченных коллекций данных.
let array = [1, 2, 3];
// Function (Функция): Блок кода, который можно вызвать для выполнения определенной задачи. Функции в JS также являются объектами.
let func = function () { return "Hi"; };

// Проверка типа данных с помощью оператора typeof (вывод в консоль)
console.log(typeof string);      // Выведет "string"
console.log(typeof number);      // Выведет "number"
console.log(typeof boolean);     // Выведет "boolean"
console.log(typeof nullValue);   // Выведет "object" (Это известная историческая ошибка в JavaScript, typeof null возвращает "object")
console.log(typeof undefinedValue); // Выведет "undefined"
console.log(typeof object);      // Выведет "object"
console.log(typeof array);       // Выведет "object" (массивы в JS считаются объектами)
console.log(typeof func);        // Выведет "function"

Операторы

JavaScript поддерживает различные типы операторов для выполнения операций над данными.

// Этот код демонстрирует различные типы операторов в JavaScript.

// Выводим заголовок в консоль
console.log("\n--- Операторы ---");

// Арифметические операторы: Используются для математических вычислений.
let sum = 5 + 3;        // Сложение: 5 + 3 = 8
let difference = 5 - 3;  // Вычитание: 5 - 3 = 2
let product = 5 * 3;     // Умножение: 5 * 3 = 15
let quotient = 10 / 2;   // Деление: 10 / 2 = 5
let remainder = 10 % 3;  // Остаток от деления: 10 / 3 = 3 с остатком 1
let power = 2 ** 3;      // Возведение в степень: 2 в степени 3 = 8
// Выводим результаты арифметических операций
console.log("Арифметика:", sum, difference, product, quotient, remainder, power);

// Операторы сравнения: Используются для сравнения значений. Возвращают true или false.
console.log("Сравнение:");
console.log("5 > 3:", 5 > 3);      // Больше: true (5 больше 3)
console.log("5 < 3:", 5 < 3);      // Меньше: false (5 не меньше 3)
console.log("5 >= 5:", 5 >= 5);     // Больше или равно: true (5 больше или равно 5)
console.log("5 <= 3:", 5 <= 3);     // Меньше или равно: false (5 не меньше или равно 3)
console.log('5 == "5":', 5 == "5");   // Равно (с приведением типов): true (строка "5" приводится к числу 5, и 5 равно 5)
console.log('5 === "5":', 5 === "5");  // Строго равно (без приведения типов): false (число 5 и строка "5" - разные типы)
console.log('5 != "5":', 5 != "5");   // Не равно (с приведением типов): false (после приведения типов 5 равно 5)
console.log('5 !== "5":', 5 !== "5");  // Строго не равно (без приведения типов): true (число 5 и строка "5" - разные типы)

// Логические операторы: Используются для работы с логическими значениями (true/false).
console.log("Логика:");
console.log("true && false:", true && false); // Логическое И (AND): true только если оба операнда true. Здесь false.
console.log("true || false:", true || false); // Логическое ИЛИ (OR): true если хотя бы один операнд true. Здесь true.
console.log("!true:", !true);         // Логическое НЕ (NOT): Инвертирует логическое значение. !true равно false.

// Инкремент (++) и декремент (--) операторы: Используются для увеличения или уменьшения значения переменной на 1.
console.log("Инкремент/Декремент:");
let i = 5;
// Постфиксный инкремент (i++): Сначала возвращает текущее значение i, затем увеличивает i на 1.
console.log("i++:", i++);  // Выведет 5
console.log("i после i++:", i);    // Теперь i равно 6
let j = 5;
// Префиксный инкремент (++j): Сначала увеличивает j на 1, затем возвращает новое значение j.
console.log("++j:", ++j);  // Выведет 6
console.log("j после ++j:", j);    // Теперь j равно 6

Строки и шаблонные строки

Строки в JavaScript можно создавать с помощью одинарных, двойных кавычек или обратных кавычек (для шаблонных строк).

// Этот код демонстрирует работу со строками и шаблонными строками в JavaScript.

// Выводим заголовок в консоль
console.log("\n--- Строки ---");

// Конкатенация строк: Соединение строк с помощью оператора +.
let firstName = "John";
let lastName = "Doe";
let fullName = firstName + " " + lastName; // Соединяем "John", пробел и "Doe"
console.log("Полное имя:", fullName); // Выведет "Полное имя: John Doe"

// Шаблонные строки (template literals): Используют обратные кавычки (`) и позволяют вставлять переменные или выражения прямо в строку с помощью ${...}.
let greeting = `Привет, ${firstName} ${lastName}!`; // Вставляем значения переменных firstName и lastName
let calculation = `2 + 2 = ${2 + 2}`; // Вставляем результат вычисления выражения 2 + 2
console.log("Шаблонная строка:", greeting); // Выведет "Шаблонная строка: Привет, John Doe!"
console.log("Вычисление в строке:", calculation); // Выведет "Вычисление в строке: 2 + 2 = 4"

// Многострочные шаблонные строки: Шаблонные строки могут содержать переводы строк, сохраняя форматирование.
let multiline = `Это первая строка.
Это вторая строка.
А это третья строка.`;
// Выводим многострочную строку. \n в console.log добавлено для переноса строки перед выводом.
console.log("Многострочная строка:\n", multiline);

Методы строк

JavaScript предоставляет множество встроенных методов для работы со строками, позволяющих выполнять различные операции, такие как поиск, замена, изменение регистра и т.д.

// Этот код демонстрирует использование различных встроенных методов строк в JavaScript.

let text = "Привет, мир!"; // Исходная строка для примеров.

// Получение длины строки: Свойство length возвращает количество символов в строке.
console.log("Длина строки:", text.length);            // Выведет 12

// Получение символа по индексу: Доступ к символу по его позиции (индексу). Индексация начинается с 0.
console.log(text[0]); // Выведет "П" (первый символ)
console.log(text.charAt(0)); // То же самое, но с использованием метода charAt(). Выведет "П".

// Поиск подстроки:
console.log("Поиск 'мир':", text.indexOf("мир"));    // Метод indexOf() возвращает индекс первого вхождения подстроки. Если не найдено, возвращает -1. Выведет 8.
console.log("Содержит 'Привет':", text.includes("Привет"));   // Метод includes() проверяет, содержит ли строка указанную подстроку. Возвращает true или false. Выведет true.

// Извлечение подстроки:
console.log(text.substring(0, 6)); // Метод substring(startIndex, endIndex) извлекает часть строки от startIndex до endIndex (не включая endIndex). Выведет "Привет".
console.log("Извлечение 'мир':", text.slice(8, 11));      // Метод slice(startIndex, endIndex) работает похоже на substring, но может принимать отрицательные индексы. Выведет "мир".

// Замена подстроки: Метод replace(searchValue, newValue) заменяет первое вхождение searchValue на newValue.
console.log("Замена 'мир':", text.replace("мир", "JavaScript")); // Выведет "Привет, JavaScript!"

// Преобразование регистра:
console.log("Верхний регистр:", text.toUpperCase()); // Метод toUpperCase() преобразует все символы строки в верхний регистр. Выведет "ПРИВЕТ, МИР!".
console.log("Нижний регистр:", text.toLowerCase()); // Метод toLowerCase() преобразует все символы строки в нижний регистр. Выведет "привет, мир!".

// Удаление пробелов в начале и конце: Метод trim() удаляет пробельные символы с обоих концов строки.
let paddedText = "   Текст с пробелами   ";
console.log("Удаление пробелов:", paddedText.trim()); // Выведет "Текст с пробелами"

// Разделение строки на массив: Метод split(separator) разбивает строку на массив подстрок, используя указанный разделитель.
console.log("Разбиение по ', ':", "яблоко,банан,киви".split(",")); // Выведет ["яблоко", "банан", "киви"]

Условные конструкции

Условные конструкции позволяют выполнять различный код в зависимости от выполнения определенных условий. Основные конструкции: if, else if, else, тернарный оператор и switch. 

// Этот код демонстрирует использование условных конструкций в JavaScript.

// Выводим заголовок в консоль
console.log("\n--- Условия ---");

// if-else: Выполняет один блок кода, если условие истинно (true), и другой блок (в else), если условие ложно (false).
let age = 20;
if (age >= 18) {
  // Этот код выполнится, если age больше или равно 18
  console.log("Вы совершеннолетний");
} else {
  // Этот код выполнится, если age меньше 18
  console.log("Вы несовершеннолетний");
}

// if-else if-else: Позволяет проверять несколько условий последовательно.
let score = 85;
if (score >= 90) {
  // Проверяем первое условие
  console.log("Оценка: Отлично");
} else if (score >= 70) {
  // Если первое условие ложно, проверяем второе
  console.log("Оценка: Хорошо");
} else if (score >= 50) {
  // Если второе условие ложно, проверяем третье
  console.log("Оценка: Удовлетворительно");
} else {
  // Если ни одно из предыдущих условий не истинно, выполняется блок else
  console.log("Оценка: Неудовлетворительно");
}

// Тернарный оператор (? :): Сокращенная форма if-else для простых условий. Синтаксис: условие ? значение_если_true : значение_если_false.
let status = age >= 18 ? "Взрослый" : "Ребенок";
console.log("Статус:", status); // Выведет "Статус: Взрослый", так как age >= 18 истинно.

// switch: Используется для выбора одного блока кода для выполнения из множества вариантов. Часто используется как альтернатива длинной цепочке if-else if.
let day = 3;
let dayName;
switch (day) {
  case 1: // Если day равно 1
    dayName = "Понедельник";
    break; // break обязателен для выхода из switch после выполнения соответствующего case
  case 2: // Если day равно 2
    dayName = "Вторник";
    break;
  case 3: // Если day равно 3
    dayName = "Среда";
    break;
  case 4: // Если day равно 4
    dayName = "Четверг";
    break;
  case 5: // Если day равно 5
    dayName = "Пятница";
    break;
  case 6: // Если day равно 6
    dayName = "Суббота";
    break;
  case 7: // Если day равно 7
    dayName = "Воскресенье";
    break;
  default: // Если значение day не совпадает ни с одним из case
    dayName = "Неизвестный день";
}
console.log("День недели:", dayName); // Выведет "День недели: Среда"

Преобразование типов

JavaScript может автоматически преобразовывать типы данных (неявное преобразование), но часто требуется явное преобразование для контроля над результатом.

// Этот код демонстрирует явное и неявное преобразование типов в JavaScript.

// Выводим заголовок в консоль
console.log("\n--- Преобразование типов ---");

// Строковое преобразование: Преобразование значения к строке.
let num = 42;
let strFromNum = String(num); // Явное преобразование числа 42 в строку "42"
let strFromBool = String(true); // Явное преобразование булева true в строку "true"
console.log("Число в строку:", strFromNum, typeof strFromNum); // Выведет "Число в строку: 42 string"
console.log("Булево в строку:", strFromBool, typeof strFromBool); // Выведет "Булево в строку: true string"

// Числовое преобразование: Преобразование значения к числу.
let str = "42";
let strFloat = "42.5";
let strInvalid = "Привет";
let numFromStr = Number(str);   // Явное преобразование строки "42" в число 42
let numFromFloat = Number(strFloat); // Явное преобразование строки "42.5" в число 42.5
let numFromInvalid = Number(strInvalid); // Явное преобразование строки "Привет" в число. Результат NaN (Not a Number), так как строка не является корректным числом.
let numFromBoolTrue = Number(true); // Явное преобразование булева true в число 1
let numFromBoolFalse = Number(false); // Явное преобразование булева false в число 0
console.log('Строка "42" в число:', numFromStr, typeof numFromStr); // Выведет 'Строка "42" в число: 42 number'
console.log('Строка "42.5" в число:', numFromFloat, typeof numFromFloat); // Выведет 'Строка "42.5" в число: 42.5 number'
console.log('Строка "Привет" в число:', numFromInvalid, typeof numFromInvalid); // Выведет 'Строка "Привет" в число: NaN number'
console.log("true в число:", numFromBoolTrue); // Выведет "true в число: 1"
console.log("false в число:", numFromBoolFalse); // Выведет "false в число: 0"

// Также существуют функции parseInt() и parseFloat() для извлечения чисел из строк, которые могут содержать нечисловые символы в конце.
console.log("parseInt('42px'):", parseInt("42px")); // parseInt() парсит строку до первого нечислового символа. Выведет 42.
console.log("parseFloat('42.5em'):", parseFloat("42.5em")); // parseFloat() парсит строку до первого нечислового символа, учитывая десятичную точку. Выведет 42.5.

// Логическое преобразование: Преобразование значения к булеву типу (true или false).
console.log("Логическое преобразование:");
console.log("Boolean(1):", Boolean(1));     // Явное преобразование числа 1 в булево. Любое ненулевое число преобразуется в true.
console.log("Boolean(0):", Boolean(0));     // Явное преобразование числа 0 в булево. 0 преобразуется в false.
console.log('Boolean(""):', Boolean(""));    // Явное преобразование пустой строки в булево. Пустая строка преобразуется в false.
console.log('Boolean("text"):', Boolean("text")); // Явное преобразование непустой строки в булево. Непустая строка преобразуется в true.
console.log("Boolean(null):", Boolean(null));  // null преобразуется в false.
console.log("Boolean(undefined):", Boolean(undefined)); // undefined преобразуется в false.
console.log("Boolean(NaN):", Boolean(NaN)); // NaN преобразуется в false.
console.log("Boolean({}):", Boolean({})); // Любой объект (даже пустой) преобразуется в true.
console.log("Boolean([]):", Boolean([])); // Любой массив (даже пустой) преобразуется в true.

Деструктуризация в JavaScript

Деструктуризация - это удобный способ "распаковывать" значения из массивов или свойства из объектов в отдельные переменные. Это делает код более читаемым и лаконичным.

Деструктуризация объектов

Деструктуризация позволяет легко извлекать свойства объекта и присваивать их новым переменным с тем же именем или другим именем.

Простой пример

Без деструктуризации вам пришлось бы обращаться к свойствам объекта через точку:

// Этот код показывает, как получить доступ к свойствам объекта без деструктуризации.

const user = {
    name: "Jacob", // Свойство name со значением "Jacob"
    age: 32        // Свойство age со значением 32
};

// Для доступа к значению свойства используем синтаксис "объект.имя_свойства"
console.log(user.name); // Выведет значение свойства name, то есть "Jacob"
console.log(user.age);  // Выведет значение свойства age, то есть 32

С деструктуризацией вы можете извлечь свойства в переменные за одну операцию:

// Этот код показывает, как использовать деструктуризацию объектов.

const user = {
    name: "Jacob", // Объект с свойствами name и age
    age: 32
};

// Деструктуризация объекта:
// Мы создаем переменные name и age и присваиваем им значения одноименных свойств объекта user.
// Синтаксис: const { имя_свойства1, имя_свойства2, ... } = объект;
const { name, age } = user;

// Теперь мы можем использовать переменные name и age напрямую
console.log(name); // Выведет "Jacob"
console.log(age);  // Выведет 32

Значения по умолчанию и переименование

При деструктуризации можно задать значения по умолчанию для свойств, которых может не быть в объекте, а также переименовать извлекаемые переменные.

// Этот код демонстрирует установку значений по умолчанию и переименование при деструктуризации объектов.

const book = {
    title: "The Last Kingdom", // Объект book с свойствами title и author
    author: "Bernard Cornwell"
};

// Деструктуризация с переименованием и значением по умолчанию:
// { title } - извлекаем свойство title в переменную с тем же именем.
// { author: bookAuthor } - извлекаем свойство author и переименовываем его в bookAuthor.
// { coverImage = "..." } - пытаемся извлечь свойство coverImage. Если его нет в объекте book, переменной coverImage будет присвоено значение по умолчанию (URL картинки).
const { title, author: bookAuthor, coverImage = "https://via.placeholder.com/640/480" } = book;

// Выводим значения извлеченных переменных
console.log(title);     // Выведет "The Last Kingdom"
console.log(bookAuthor); // Выведет "Bernard Cornwell" (используем новое имя переменной)
console.log(coverImage); // Выведет значение по умолчанию, так как свойства coverImage нет в объекте book. Выведет "https://via.placeholder.com/640/480".

Деструктуризация в циклах

Деструктуризация очень удобна при переборе массивов, содержащих объекты, например, в цикле for...of. 

// Этот код показывает, как использовать деструктуризацию объектов прямо в цикле for...of.

// Массив объектов, где каждый объект представляет книгу.
const books = [
    { title: "The Last Kingdom", author: "Bernard Cornwell", rating: 8.38 },
    { title: "Beside Still Waters", author: "Robert Sheckley", rating: 8.51 }
];

// Перебираем массив books с помощью цикла for...of.
// В каждой итерации цикла, вместо того чтобы получать весь объект книги (например, 'book'),
// мы сразу деструктурируем его, извлекая свойства title, author и rating в отдельные переменные.
for (const { title, author, rating } of books) {
    // Используем извлеченные переменные для вывода информации о каждой книге.
    console.log(`Книга: ${title}, Автор: ${author}, Рейтинг: ${rating}`);
}
// Результат будет:
// Книга: The Last Kingdom, Автор: Bernard Cornwell, Рейтинг: 8.38
// Книга: Beside Still Waters, Автор: Robert Sheckley, Рейтинг: 8.51

Деструктуризация параметров функции

Деструктуризация позволяет извлекать свойства объекта, переданного в качестве аргумента функции, прямо в списке параметров функции.

// Этот код показывает, как использовать деструктуризацию объектов в параметрах функции.

// Объявляем функцию printUserInfo.
// В параметрах функции мы ожидаем объект. Вместо того чтобы объявлять один параметр (например, 'user')
// и затем внутри функции деструктурировать его, мы делаем это прямо в сигнатуре функции:
// { name, age, hobby } - это деструктуризация входящего объекта.
// Это означает, что функция ожидает объект, который имеет свойства name, age и hobby,
// и эти свойства будут автоматически извлечены в локальные переменные name, age и hobby внутри функции.
function printUserInfo({ name, age, hobby }) {
    // Используем извлеченные переменные name, age и hobby внутри функции.
    console.log(`Name: ${name}, Age: ${age}, Hobby: ${hobby}`);
}

// Вызываем функцию printUserInfo, передавая ей объект.
// Свойства этого объекта (name, age, hobby) будут деструктурированы в параметры функции.
printUserInfo({
    name: "Alice",
    age: 25,
    hobby: "dancing"
});
// Выведет: Name: Alice, Age: 25, Hobby: dancing

Глубокая деструктуризация

Если объект содержит вложенные объекты, можно деструктурировать их свойства на любом уровне вложенности.

// Этот код демонстрирует глубокую деструктуризацию объектов.

// Объект user с вложенным объектом stats.
const user = {
    name: "Jacques Gluke",
    tag: "jgluke",
    stats: { // Вложенный объект stats
        followers: 5603,
        views: 4827,
        likes: 1308,
    },
};

// Глубокая деструктуризация:
// { name, tag } - извлекаем свойства name и tag из верхнего уровня объекта user.
// { stats: { followers, views, likes } } - это деструктуризация вложенного объекта.
// stats: - указывает, что мы хотим работать со свойством stats объекта user.
// { followers, views, likes } - внутри фигурных скобок мы указываем свойства, которые хотим извлечь из объекта stats.
// В результате будут созданы переменные name, tag, followers, views, likes.
const { name, tag, stats: { followers, views, likes } } = user;

// Выводим значения извлеченных переменных.
console.log(name);     // Выведет "Jacques Gluke"
console.log(tag);      // Выведет "jgluke"
console.log(followers); // Выведет 5603 (извлечено из user.stats.followers)
console.log(views);     // Выведет 4827 (извлечено из user.stats.views)
console.log(likes);     // Выведет 1308 (извлечено из user.stats.likes)

// Обратите внимание: переменная stats НЕ будет создана при такой деструктуризации.
// Если вам нужна сама переменная stats, нужно деструктурировать ее отдельно:
// const { name, tag, stats } = user; // Теперь stats - это объект { followers: 5603, views: 4827, likes: 1308 }
// const { followers, views, likes } = stats; // И затем деструктурировать stats

Деструктуризация массивов

Деструктуризация массивов позволяет извлекать элементы массива и присваивать их переменным по их позиции (индексу). Используются квадратные скобки [].

Простой пример

Без деструктуризации вам пришлось бы обращаться к элементам массива по их индексу:

// Этот код показывает, как получить доступ к элементам массива без деструктуризации.

const color = [200, 255, 100]; // Массив с тремя элементами (значения RGB)

// Для доступа к элементу массива используем синтаксис "массив[индекс]". Индексы начинаются с 0.
const red = color[0];   // Получаем первый элемент (индекс 0)
const green = color[1]; // Получаем второй элемент (индекс 1)
const blue = color[2];  // Получаем третий элемент (индекс 2)

// Используем полученные значения
console.log(`rgb(${red}, ${green}, ${blue})`); // Выведет "rgb(200, 255, 100)"

С деструктуризацией вы можете извлечь элементы в переменные за одну операцию:

// Этот код показывает, как использовать деструктуризацию массивов.

const color = [200, 255, 100]; // Массив с элементами

// Деструктуризация массива:
// Мы создаем переменные red, green, blue и присваиваем им значения элементов массива color
// по их позиции. red получит значение первого элемента, green - второго, blue - третьего.
// Синтаксис: const [ переменная1, переменная2, ... ] = массив;
const [red, green, blue] = color;

// Теперь мы можем использовать переменные red, green, blue напрямую
console.log(`rgb(${red}, ${green}, ${blue})`); // Выведет "rgb(200, 255, 100)"

Значения по умолчанию

Если при деструктуризации массива переменных больше, чем элементов в массиве, переменным, для которых нет соответствующего элемента, будет присвоено значение undefined. Вы можете задать значения по умолчанию.

// Этот код демонстрирует установку значений по умолчанию при деструктуризации массивов.

const color = [200, 100, 255]; // Массив с тремя элементами

// Деструктуризация массива со значением по умолчанию:
// Мы извлекаем red, green, blue как раньше.
// Для переменной alfa мы указываем значение по умолчанию = 0.3.
// Так как в массиве color нет четвертого элемента, переменной alfa будет присвоено значение 0.3.
const [red, green, blue, alfa = 0.3] = color;

// Используем извлеченные значения
console.log(`rgba(${red}, ${green}, ${blue}, ${alfa})`); // Выведет "rgba(200, 100, 255, 0.3)"

Оператор rest

Оператор rest (...) при деструктуризации массива позволяет собрать оставшиеся элементы массива в новый массив.

// Этот код демонстрирует использование оператора rest (...) при деструктуризации массивов.

const color = [200, 255, 100, 50, 150]; // Массив с несколькими элементами

// Деструктуризация с оператором rest:
// [red, ...otherColors] - извлекаем первый элемент в переменную red.
// ...otherColors - оператор rest собирает ВСЕ ОСТАВШИЕСЯ элементы массива color
// после первого элемента в новый массив под названием otherColors.
const [red, ...otherColors] = color;

// Выводим извлеченные значения
console.log(red);        // Выведет 200 (первый элемент)
console.log(otherColors); // Выведет [255, 100, 50, 150] (новый массив с оставшимися элементами)

Пропуск значений

Чтобы пропустить элементы массива при деструктуризации, просто оставляйте пустые места между запятыми.

// Этот код показывает, как пропустить элементы при деструктуризации массива.

const rgb = [200, 100, 255]; // Массив RGB значений

// Деструктуризация с пропуском значений:
// [ , , blue ] - мы используем запятые, чтобы пропустить первые два элемента массива.
// Переменная blue получит значение третьего элемента массива (индекс 2).
const [, , blue] = rgb;

// Выводим извлеченное значение
console.log(`Blue: ${blue}`); // Выведет "Blue: 255"

Деструктуризация параметров функции

Как и с объектами, можно деструктурировать массив, переданный в качестве аргумента, прямо в параметрах функции. 

// Этот код показывает, как использовать деструктуризацию массивов в параметрах функции.

// Объявляем функцию printFruits.
// В параметрах функции мы ожидаем массив. Мы деструктурируем его прямо в сигнатуре функции:
// [firstFruit, secondFruit, thirdFruit] - это деструктуризация входящего массива.
// Функция ожидает массив, и его первый, второй и третий элементы будут автоматически извлечены
// в локальные переменные firstFruit, secondFruit и thirdFruit соответственно.
function printFruits([firstFruit, secondFruit, thirdFruit]) {
    // Используем извлеченные переменные внутри функции.
    console.log(firstFruit, secondFruit, thirdFruit);
}

// Вызываем функцию printFruits, передавая ей массив.
// Элементы этого массива будут деструктурированы в параметры функции.
printFruits(["apple", "banana", "orange"]); // Выведет: apple banana orange

// Если передать массив с меньшим количеством элементов, переменные, для которых нет соответствия, будут undefined.
// printFruits(["grape", "kiwi"]); // Выведет: grape kiwi undefined

Визуальные материалы

Схемы помогают лучше понять синтаксис деструктуризации.

Синтаксис деструктуризации


Деструктуризация
│
├── Объекты
│   ├── Базовый синтаксис: const { имя_свойства } = объект
│   ├── Переименование: const { имя_свойства: новое_имя } = объект
│   ├── Значение по умолчанию: const { имя_свойства = значение } = объект
│   ├── Глубокая деструктуризация: const { свойство: { вложенное } } = объект
│   └── Rest: const { свойство, ...остальное } = объект
│
└── Массивы
    ├── Базовый синтаксис: const [переменная1, переменная2] = массив
    ├── Пропуск: const [, , переменная3] = массив
    ├── Значение по умолчанию: const [переменная1 = значение] = массив
    └── Rest: const [первая, ...остальные] = массив
