Практические упражнения

Закрепите свои знания, выполнив следующие упражнения. Попробуйте решить их самостоятельно перед тем, как смотреть решения.

Упражнения по основам

  1. Объявите переменную для хранения вашего имени и выведите ее в консоль.
  2. Создайте две числовые переменные, сложите их и выведите результат.
  3. Напишите условную конструкцию, которая проверяет, является ли число четным или нечетным.
  4. Создайте строку и выведите ее длину и первый символ.
console.log("===== Упражнения по основам =====");

// 1. Имя
let myName = "ВашеИмя";
console.log(myName);

// 2. Сложение
let num1 = 15;
let num2 = 7;
console.log(num1 + num2);

// 3. Чет/Нечет
let checkNum = 10;
if (checkNum % 2 === 0) {
    console.log(checkNum + " - четное");
} else {
    console.log(checkNum + " - нечетное");
}

// 4. Строка
let myString = "JavaScript";
console.log("Длина:", myString.length);
console.log("Первый символ:", myString[0]);

Упражнения по массивам

  1. Создайте массив ваших любимых фруктов. Добавьте новый фрукт в конец массива.
  2. Используя метод `map`, создайте новый массив, где каждый элемент исходного числового массива умножен на 2.
  3. Используя метод `filter`, создайте новый массив, содержащий только строки длиннее 5 символов из исходного массива строк.
  4. Найдите сумму всех чисел в массиве с помощью метода `reduce`.
console.log("\n===== Упражнения по массивам =====");

// 1. Фрукты
let favFruits = ["Яблоко", "Банан"];
favFruits.push("Манго");
console.log(favFruits);

// 2. Map
const numbersMap = [1, 2, 3, 4];
const doubled = numbersMap.map(n => n * 2);
console.log(doubled);

// 3. Filter
const words = ["Привет", "Мир", "JavaScript", "Кот", "Программирование"];
const longWords = words.filter(word => word.length > 5);
console.log(longWords);

// 4. Reduce
const numbersReduce = [5, 10, 15, 20];
const sum = numbersReduce.reduce((acc, current) => acc + current, 0);
console.log(sum);

Упражнения по функциям

  1. Напишите функцию, которая принимает два числа и возвращает их сумму.
  2. Создайте функцию, которая принимает массив чисел и возвращает наибольшее число в этом массиве.
  3. Напишите стрелочную функцию, которая принимает строку и возвращает ее в верхнем регистре.
  4. Создайте функцию, которая возвращает другую функцию (замыкание). Внутренняя функция должна использовать переменную из внешней функции.
console.log("\n===== Упражнения по функциям =====");

// 1. Сумма
function sumNumbers(a, b) {
    return a + b;
}
console.log(sumNumbers(5, 3));

// 2. Максимум в массиве
function findMax(arr) {
    if (arr.length === 0) return undefined;
    return Math.max(...arr); // Используем spread оператор
}
console.log(findMax([1, 5, 2, 9, 3]));

// 3. Верхний регистр (стрелочная)
const toUpper = str => str.toUpperCase();
console.log(toUpper("hello"));

// 4. Замыкание
function createGreeter(greeting) {
    return function(name) {
        console.log(`${greeting}, ${name}!`);
    }
}
const greetHello = createGreeter("Здравствуйте");
greetHello("Ирина");

Упражнения по циклам

  1. Используя цикл `for`, выведите числа от 1 до 10.
  2. Используя цикл `while`, найдите факториал числа 5.
  3. Переберите массив строк с помощью `for...of` и выведите каждую строку.
  4. Переберите свойства объекта с помощью `for...in` и выведите их ключи и значения.
console.log("\n===== Упражнения по циклам =====");

// 1. Цикл for 1-10
for (let i = 1; i <= 10; i++) {
    console.log(i);
}

// 2. Факториал (while)
let numFact = 5;
let resultFact = 1;
let iFact = numFact;
while (iFact > 1) {
    resultFact *= iFact;
    iFact--;
}
console.log(`Факториал ${numFact}:`, resultFact);

// 3. for...of
const loopWords = ["Один", "Два", "Три"];
for (const word of loopWords) {
    console.log(word);
}

// 4. for...in
const loopPerson = { name: "Олег", age: 40 };
for (const key in loopPerson) {
    console.log(`${key}: ${loopPerson[key]}`);
}

Упражнения по объектам

  1. Создайте объект, представляющий книгу (свойства: title, author, year).
  2. Добавьте метод к объекту книги, который выводит информацию о книге в формате "Название - Автор (Год)".
  3. Создайте массив объектов книг.
  4. Напишите функцию, которая принимает массив книг и возвращает массив названий книг, опубликованных после 2000 года.
console.log("\n===== Упражнения по объектам =====");

// 1, 2. Объект Книга
const book = {
    title: "Мастер и Маргарита",
    author: "М. Булгаков",
    year: 1967,
    getInfo: function() {
        console.log(`${this.title} - ${this.author} (${this.year})`);
    }
};
book.getInfo();

// 3. Массив книг
const library = [
    book,
    { title: "1984", author: "Дж. Оруэлл", year: 1949 },
    { title: "Гарри Поттер и философский камень", author: "Дж. Роулинг", year: 1997 },
    { title: "Код да Винчи", author: "Д. Браун", year: 2003 }
];

// 4. Фильтр по году
function getModernBooks(books) {
    return books
        .filter(b => b.year > 2000)
        .map(b => b.title);
}
console.log("Книги после 2000:", getModernBooks(library));
