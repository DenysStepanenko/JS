Памятка по работе с DOM

В этом разделе собраны основные методы и свойства для работы с DOM, которые мы изучили. Каждый метод сопровождается кратким описанием и примером.

Поиск DOM-элементов

Методы для поиска элементов на странице:

  • element.querySelector(selector) — возвращает первый элемент, соответствующий CSS-селектору selector внутри element.
  • element.querySelectorAll(selector) — возвращает все элементы (в виде NodeList), соответствующие CSS-селектору selector внутри element.

Пример:

console.log("===== Поиск DOM-элементов =====");

// Находим первый заголовок 

const heading = document.querySelector("h1");
console.log("Первый заголовок:", heading);

// Находим все элементы с классом 'item'
const items = document.querySelectorAll(".item");
console.log("Все элементы с классом 'item':", items);

Навигация по DOM

DOM представляет собой иерархическое дерево, где элементы связаны как родитель, дети, братья и сестры. Вот основные свойства для навигации:

  • element.parentElement — возвращает родительский элемент.
  • element.children — возвращает коллекцию дочерних элементов (только элементы, без текстовых узлов).
  • element.firstElementChild — возвращает первый дочерний элемент.
  • element.lastElementChild — возвращает последний дочерний элемент.
  • element.nextElementSibling — возвращает следующий соседний элемент.
  • element.previousElementSibling — возвращает предыдущий соседний элемент.

Пример:

console.log("===== Навигация по DOM =====");

// Создаём структуру
const ul = document.createElement("ul");
const li1 = document.createElement("li");
li1.textContent = "Элемент 1";
const li2 = document.createElement("li");
li2.textContent = "Элемент 2";
ul.append(li1, li2);

// Находим родителя первого

console.log("Родитель li1:", li1.parentElement); // 
      

// Находим детей 
        
console.log("Дети ul:", ul.children); // [li1, li2]

// Находим первый и последний дочерний элемент
console.log("Первый ребёнок ul:", ul.firstElementChild); // li1
console.log("Последний ребёнок ul:", ul.lastElementChild); // li2

// Находим соседей li1
console.log("Следующий сосед li1:", li1.nextElementSibling); // li2
console.log("Предыдущий сосед li2:", li2.previousElementSibling); // li1

    • Свойства DOM-элемента

    Свойства для чтения и изменения содержимого, стилей и атрибутов:

    • element.textContent — содержит только текст внутри элемента, игнорируя HTML-теги. Можно читать и изменять.
    • element.innerHTML — содержит HTML-код внутри элемента (включая теги). Можно читать и изменять.
    • element.style — объект, содержащий встроенные стили элемента (свойства в camelCase).
    • element.hasAttribute(name) — проверяет, есть ли атрибут name у элемента.
    • element.getAttribute(name) — возвращает значение атрибута name.
    • element.setAttribute(name, value) — устанавливает атрибут name со значением value.
    • element.removeAttribute(name) — удаляет атрибут name.
    • element.dataset — объект для работы с data-атрибутами (например, data-action → dataset.action).

    Пример:

    console.log("===== Свойства DOM-элемента =====");

// Создаём элемент
const div = document.createElement("div");
div.innerHTML = "
    Привет
    ";

// Читаем textContent
console.log("textContent:", div.textContent); // "Привет"

// Читаем innerHTML
console.log("innerHTML:", div.innerHTML); // "
    Привет
    "

// Меняем стиль
div.style.backgroundColor = "lightblue";
console.log("style.backgroundColor:", div.style.backgroundColor); // "lightblue"

// Работа с атрибутами
div.setAttribute("data-type", "info");
console.log("Есть ли атрибут data-type?", div.hasAttribute("data-type")); // true
console.log("Значение data-type:", div.getAttribute("data-type")); // "info"
console.log("dataset.type:", div.dataset.type); // "info"

// Удаляем атрибут
div.removeAttribute("data-type");
console.log("Атрибут data-type удалён:", !div.hasAttribute("data-type")); // true

    CSS-классы на DOM-элементах

    Методы для управления классами через classList:

    • element.classList.contains(className) — проверяет, есть ли класс className.
    • element.classList.add(className) — добавляет класс className.
    • element.classList.remove(className) — удаляет класс className.
    • element.classList.toggle(className) — переключает класс className.
    • element.classList.replace(oldClassName, newClassName) — заменяет класс oldClassName на newClassName.

    Пример:

    console.log("===== Управление CSS-классами =====");

// Создаём элемент
const box = document.createElement("div");
box.classList.add("box");

// Проверяем класс
console.log("Есть ли класс 'box'?", box.classList.contains("box")); // true

// Добавляем класс
box.classList.add("active");
console.log("Классы после добавления:", box.classList); // ["box", "active"]

// Переключаем класс
box.classList.toggle("active");
console.log("Классы после toggle:", box.classList); // ["box"]

// Заменяем класс
box.classList.replace("box", "new-box");
console.log("Классы после замены:", box.classList); // ["new-box"]

    Создание и удаление DOM-элементов

    Методы для создания, добавления и удаления элементов:

    • document.createElement(tagName) — создаёт новый HTML-элемент с тегом tagName.
    • element.append(el) — добавляет el в конец списка дочерних элементов element.
    • element.prepend(el) — добавляет el в начало списка дочерних элементов element.
    • element.remove() — удаляет элемент element из DOM-дерева.
    • element.insertAdjacentHTML(position, string) — вставляет HTML-строку string в позицию position.

    Пример:

    console.log("===== Создание и удаление элементов =====");

// Создаём элемент
const newDiv = document.createElement("div");
newDiv.textContent = "Новый элемент";

// Добавляем в начало body
document.body.prepend(newDiv);
console.log("Добавлен элемент:", document.body.firstElementChild);

// Добавляем HTML через insertAdjacentHTML
newDiv.insertAdjacentHTML("beforeend", "
    Внутри div
    ");
console.log("HTML внутри div:", newDiv.innerHTML);

// Удаляем элемент
newDiv.remove();
console.log("Элемент удалён:", document.body.firstElementChild);

    События

    Методы и события для обработки действий пользователя:

    • element.addEventListener(event, handler) — добавляет обработчик handler для события event.
    • element.removeEventListener(event, handler) — удаляет обработчик handler для события event.
    • click — срабатывает при клике мышью.
    • keydown — срабатывает при нажатии клавиши.
    • keyup — срабатывает при отпускании клавиши.
    • submit — срабатывает при отправке формы.
    • change — срабатывает после изменения значения элемента формы.
    • input — срабатывает при каждом изменении текстового поля.
    • focus — срабатывает, когда элемент получает фокус.
    • blur — срабатывает, когда элемент теряет фокус.

    Пример:

    console.log("===== Работа с событиями =====");

// Создаём кнопку
const btn = document.createElement("button");
btn.textContent = "Нажми меня";

// Создаём обработчик
const handleClick = () => {
    console.log("Кнопка нажата!");
};

// Добавляем слушатель
btn.addEventListener("click", handleClick);

// Пример события input
const input = document.createElement("input");
input.addEventListener("input", event => {
    console.log("Введено:", event.target.value);
});

// Пример события focus
input.addEventListener("focus", () => {
    console.log("Поле в фокусе!");
});

// Пример события blur
input.addEventListener("blur", () => {
    console.log("Поле потеряло фокус!");
});

// Удаляем слушатель
btn.removeEventListener("click", handleClick);
console.log("Слушатель клика удалён");

// Пример события keydown
document.addEventListener("keydown", event => {
    console.log("Клавиша нажата:", event.key, "Код:", event.code);
});

    Интерактивный пример: TODO-список

    Попробуй добавить задачи, отметить их как выполненные или удалить. Этот пример использует все изученные методы и события.

      Код примера:

      console.log("===== Интерактивный TODO-список =====");

// Находим элементы
const form = document.querySelector(".todo-form");
const input = document.querySelector(".todo-input");
const list = document.querySelector(".todo-list");

// Проверяем, найдены ли элементы
if (!form || !input || !list) {
    console.log("Элементы не найдены");
} else {
    // Слушаем отправку формы
    form.addEventListener("submit", event => {
        event.preventDefault(); // Отменяем перезагрузку

        const taskText = input.value.trim(); // Получаем текст задачи
        if (taskText === "") {
            console.log("Введите задачу!");
            return;
        }

        // Создаём новый элемент списка
        const li = document.createElement("li");
        li.classList.add("todo-item");

        // Создаём span для текста задачи
        const taskSpan = document.createElement("span");
        taskSpan.textContent = taskText;

        // Создаём кнопку удаления
        const deleteBtn = document.createElement("button");
        deleteBtn.textContent = "Удалить";

        // Добавляем элементы в 
    • 
        li.append(taskSpan, deleteBtn);

        // Слушаем клик для отметки задачи как выполненной
        taskSpan.addEventListener("click", () => {
            li.classList.toggle("completed");
        });

        // Слушаем клик для удаления задачи
        deleteBtn.addEventListener("click", () => {
            li.remove();
        });

        // Добавляем задачу в список
        list.append(li);

        // Очищаем поле ввода и возвращаем фокус
        input.value = "";
        input.focus();
    });

    // Слушаем ввод в реальном времени
    input.addEventListener("input", event => {
        if (event.currentTarget.value.length > 20) {
            input.style.borderColor = "red";
        } else {
            input.style.borderColor = "";
        }
    });

    // Подсвечиваем поле при фокусе
    input.addEventListener("focus", () => {
        input.style.borderColor = "blue";
    });

    // Проверяем при потере фокуса
    input.addEventListener("blur", () => {
        if (input.value.trim() === "") {
            input.style.borderColor = "";
        }
    });
}
